 import java.util.*;

class Solution {
    public int solution(int N) {
        // write your code in Java SE 8
        List<Integer> result = new ArrayList();
        int total = 0;
        int counter = 1;
        int next = 0;
        int rr = 0;
        int max = 0;

        while(N != 0){
            
        int r = N%2;
        N = N/2;

        if (r == 1){
            result.add(counter);
            }
        counter++;
        }
        
        Collections.reverse(result);

        for (Integer i : result) { 
            if (0 == next) {
                next = i;
                continue;
            } 
            rr = next - i;
            next = i;
            if (rr >= max) {
                max = rr; 
            }
        }
        
        if(max == 0) {return 0;} else {return max-1;}
    }
}
